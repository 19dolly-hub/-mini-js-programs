The function to find Maximum number


function findMax(arr) {
    let maxNum = arr[0];
    for(let i=0; i<arr.length; i++) {
        if(maxNum < arr[i]) {
            maxNum = arr[i];
        }
        else {
            maxNum = maxNum;
        }
    }
    return maxNum;
}
