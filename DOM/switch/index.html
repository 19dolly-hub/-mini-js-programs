Light Switch

 const room = document.querySelector("body");
 const btn = document.querySelector(".btnOff");
 let tap = 0;

 btn.addEventListener("click", () => {

   if (tap == 0) {
     room.style.backgroundColor = "#dadada";
     tap = 1;
   } else {
     room.style.backgroundColor = "black";
     tap = 0;
   }

 })
